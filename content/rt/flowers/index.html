Got me so high, and then she dropped me (Maroon five)

library(tidyverse)


x = seq(from=-5, to=5, by = 0.005) %>%
  expand.grid(x=., y=.) 
p = x %>%
  ggplot() +
  aes(x=(pi*x + sin(x*pi)), 
      y=(y*pi * sin(x*pi)), 
      color = abs(y)) + 
  geom_point(alpha=.01, shape=20, size=0)+
 scale_color_distiller(palette = "YlGn", 
 direction = 1) +
  theme_void()+
  theme(legend.position = "none" ) +
  coord_polar(); p
